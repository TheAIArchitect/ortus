53 std::string FileUtils::readFullFile(const std::string &fileName){
54  std::ifstream ifs(fileName, std::ios::in | std::ios::binary | std::ios::ate);
55  if(!ifs.is_open())
56  throw std::runtime_error("Couldn't open " + fileName);

96 size_t FileUtils::lineCount(const std::string &fileName){
97  const char *fname = fileName.c_str();
98  static const auto BUF_SIZE = 16*1024;
99  int fd = open(fname, O_RDONLY);
100  if(fd == -1){
101  throw std::runtime_error("Couldn't open " + std::string(fname));
102  }
103 
104  char buf[BUF_SIZE + 1];
105  size_t lines = 0;
106 
107  while(size_t bytes_read = read(fd, buf, BUF_SIZE)){
108  if(bytes_read == (size_t) -1)
109  throw std::runtime_error("Read Failure" + std::string(fname));
110 
111  if(!bytes_read)
112  break;
113 
114  for(char *p = buf; (p = (char *) memchr(p, '\n', (buf + bytes_read) - p)); ++p)
115  ++lines;
116  }
117  close(fd);
118  return lines;
119 }

127 std::vector<std::string> FileUtils::getFilesInDir(const std::string &dirPath, const std::string &ext){
128  std::vector<std::string> files;
129 
130  DIR *dir;
131  struct dirent *ent;
132 
133  if ((dir = opendir(dirPath.c_str())) != NULL) {
134  // print all the files and directories within directory
135  while ((ent = readdir(dir)) != NULL) {
136  if(isFile(std::string(dirPath + "/" + ent->d_name))){
137  // Only add files with the requested file extension
138  if(ext != ""){
139  // Note: This can easily be fooled
140  std::string f_name(ent->d_name);
141  if(endsWithString(f_name, ext))
142  files.push_back(f_name);
143  }
144  // Don't care what the file extension is
145  else{
146  files.emplace_back(ent->d_name);
147  }
148  }
149  }
150  closedir(dir);
151  }
152  else {
153  // could not open directory
154  throw std::runtime_error("Could not open dir " + dirPath);
155  }
156  return files;
157 }
