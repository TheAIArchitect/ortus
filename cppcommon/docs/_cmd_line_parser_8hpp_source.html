32  CmdLineParser(int argc, const char **argv){
33  for(auto i = 1; i < argc; ++i){
34  // Extra string processing to deal with -wworld type flags where the variable and flag
35  // are not separated by a space
36  std::string arg{argv[i]};
37  if(arg[0] == '-'){
38  // Check if a space already exists between flag and variable
39  if(arg.length() > 2 && arg[2] != ' '){
40  // Pull the flag out of the string to push onto the vector
41  std::string flag{arg.substr(0,2)};
42  // Pull the variable out of the string to push onto the vector
43  std::string var{arg.substr(2)};
44 
45  // Add the flag and the variable to the vector and continue the loop
46  args_.push_back(flag);
47  args_.push_back(var);
48  continue;
49  }
50  }
51  args_.emplace_back(argv[i]);
52  }
53  }
54 
59  bool optExists(const std::string &opt_ID){
60  return std::find(std::begin(args_), std::end(args_), opt_ID) != std::end(args_);
61  }

69  std::string getOptValue(const std::string &opt_ID){
70  std::vector<std::string>::const_iterator it;
71  it = std::find(std::begin(args_), std::end(args_), opt_ID);
72  if(it != std::end(args_) && ++it != std::end(args_))
73  return *it;
74  return "";
75  }
