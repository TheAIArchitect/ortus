42 std::string StrUtils::removeCharFromStr(const std::string &s, const char ch){
43  std::string s_ch{ch};
44 
45  if(s.length() == 0)
46  return "";
47  if(s.length() == 1){
48  if(s_ch == s)
49  return "";
50  return s;
51  }
52 
53  auto end = s.length() - 1;
54  std::string test_s = s.substr(0, 1);
55  std::string rest = s.substr(1, end);

89 std::string StrUtils::trim(const std::string &s){
90  if(s.find_first_not_of("\t\n\a\b\f\r\v ") == std::string::npos)
91  return "";
92 
93  std::string trimmed = s;
94  // Removing tab chars
95  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), '\t'), std::end(trimmed));
96  // Removing new line chars
97  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), '\n'), std::end(trimmed));
98  // Removing 'audible bell'
99  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), '\a'), std::end(trimmed));
100  // Removing backspace
101  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), '\b'), std::end(trimmed));
102  // Removing form-feed
103  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), '\f'), std::end(trimmed));
104  // Removing carriage return
105  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), '\r'), std::end(trimmed));
106  // Removing vertical tab
107  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), '\v'), std::end(trimmed));
108 
109  auto first_char = trimmed.find_first_not_of(" ");
110  auto last_char = trimmed.size() - first_char;
111  std::string no_leading_ws = trimmed.substr(first_char, last_char);
112  // Resetting first_char to 0;
113  first_char = 0;
114  last_char = no_leading_ws.find_last_not_of(" ");
115  std::string all_trimmed = no_leading_ws.substr(first_char, last_char + 1);
116  return all_trimmed;
117 }
